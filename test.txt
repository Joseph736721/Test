-- FONCTION REJOIN AVANCÉE (SERVER HOP)
-- ==========================================
local function TPReturner()
    local Site
    if foundAnything == "" then
        Site = HttpService:JSONDecode(game:HttpGet('https://games.roblox.com/v1/games/' .. PlaceID .. '/servers/Public?sortOrder=Asc&limit=100'))
    else
        Site = HttpService:JSONDecode(game:HttpGet('https://games.roblox.com/v1/games/' .. PlaceID .. '/servers/Public?sortOrder=Asc&limit=100&cursor=' .. foundAnything))
    end
    
    if Site.nextPageCursor and Site.nextPageCursor ~= "null" and Site.nextPageCursor ~= nil then
        foundAnything = Site.nextPageCursor
    end
    
    local num = 0
    for _, v in pairs(Site.data) do
        local Possible = true
        local ID = tostring(v.id)
        if tonumber(v.maxPlayers) > tonumber(v.playing) and ID ~= game.JobId then
            for _, Existing in pairs(AllIDs) do
                if num ~= 0 then
                    if ID == tostring(Existing) then
                        Possible = false
                    end
                else
                    if tonumber(actualHour) ~= tonumber(Existing) then
                        pcall(function()
                            delfile("NotSameServers.json")
                            AllIDs = {}
                            table.insert(AllIDs, actualHour)
                            pcall(function() writefile("NotSameServers.json", HttpService:JSONEncode(AllIDs)) end)
                        end)
                    end
                end
                num = num + 1
            end
            
            if Possible == true then
                table.insert(AllIDs, ID)
                task.wait()
                pcall(function()
                    writefile("NotSameServers.json", HttpService:JSONEncode(AllIDs))
                    task.wait()
                    TeleportService:TeleportToPlaceInstance(PlaceID, ID, LocalPlayer)
                end)
                task.wait(4)
            end
        end
    end
end

local function serverHop()
    setStatus("Server Hopping...")
    print("Tentative de Server Hop intelligent...")
    
    if queue_on_teleport then
        queue_on_teleport([[loadstring(game:HttpGet('https://raw.githubusercontent.com/Joseph736721/Test/refs/heads/main/test.txt'))()]])
    end

    pcall(function()
        AllIDs = HttpService:JSONDecode(readfile("NotSameServers.json"))
    end)
    if #AllIDs == 0 then
        table.insert(AllIDs, actualHour)
        pcall(function() writefile("NotSameServers.json", HttpService:JSONEncode(AllIDs)) end)
    end

    while task.wait(1) do
        pcall(function()
            TPReturner()
            if foundAnything ~= "" then
                TPReturner()
            end
        end)
    end
end

local function clickButton(guiPath)
    if guiPath and guiPath:IsA("GuiButton") then
        if firesignal then
            firesignal(guiPath.Activated)
            firesignal(guiPath.MouseButton1Click)
        else
            guiPath:Activate()
        end
        return true
    end
    return false
end

local function runInitialSetup()
    setStatus("Init : Attente 10s")
    task.wait(10)

    local playerGui = LocalPlayer:WaitForChild("PlayerGui")

    -- 1. Cliquer sur Aux
    setStatus("Init : Clic Aux")
    local auxButton = playerGui:FindFirstChild("OnboardingGUI")
        and playerGui.OnboardingGUI:FindFirstChild("Onboarding")
        and playerGui.OnboardingGUI.Onboarding:FindFirstChild("Main")
        and playerGui.OnboardingGUI.Onboarding.Main:FindFirstChild("Aux")

    if not auxButton or not clickButton(auxButton) then serverHop() return end
    task.wait(2)

    -- 2. Slot 1
    setStatus("Init : Clic Slot 1")
    local slotListMain = playerGui:FindFirstChild("LoadSaveGUI")
        and playerGui.LoadSaveGUI:FindFirstChild("SlotList")
        and playerGui.LoadSaveGUI.SlotList:FindFirstChild("Main")

    local slotFound = false
    if slotListMain then
        for _, descendant in ipairs(slotListMain:GetDescendants()) do
            if descendant.Name == "SlotName" and descendant.Text == "Slot 1" then
                local parentButton = descendant.Parent
                if parentButton and parentButton:IsA("TextButton") then
                    clickButton(parentButton)
                    slotFound = true
                    break
                end
            end
        end
    end
    if not slotFound then serverHop() return end
    task.wait(2)

    -- 3. Load
    setStatus("Init : Clic Load")
    local loadButton = playerGui:FindFirstChild("LoadSaveGUI")
        and playerGui.LoadSaveGUI:FindFirstChild("SlotInfo")
        and playerGui.LoadSaveGUI.SlotInfo:FindFirstChild("Main")
        and playerGui.LoadSaveGUI.SlotInfo.Main:FindFirstChild("Load")

    if not loadButton or not clickButton(loadButton) then serverHop() return end
    task.wait(5)

    -- 4. SelectPurchase
    setStatus("Init : Terrain P1")
    local selectPurchaseButton = playerGui:FindFirstChild("PropertyPurchasingGUI")
        and playerGui.PropertyPurchasingGUI:FindFirstChild("SelectPurchase")
        and playerGui.PropertyPurchasingGUI.SelectPurchase:FindFirstChild("Purchase")

    if not selectPurchaseButton or not clickButton(selectPurchaseButton) then serverHop() return end
    task.wait(2)

    -- 5. ConfirmPurchase
    setStatus("Init : Terrain P2")
    local confirmPurchaseButton = playerGui:FindFirstChild("PropertyPurchasingGUI")
        and playerGui.PropertyPurchasingGUI:FindFirstChild("ConfirmPurchase")
        and playerGui.PropertyPurchasingGUI.ConfirmPurchase:FindFirstChild("Purchase")

    if not confirmPurchaseButton or not clickButton(confirmPurchaseButton) then serverHop() return end
    
    setStatus("Initialisation Ok !")
    task.wait(5)
    
    setStatus("Lancement du Script...")
    task.wait(1)
    isInitDone = true
end

local function listenToDamage(character)
    local humanoid = character:WaitForChild("Humanoid", 10)
    if humanoid then
        local lastHealth = humanoid.Health
        humanoid.HealthChanged:Connect(function(currentHealth)
            if currentHealth < lastHealth and not isPausedForDamage and _G.FarmEnabled then
                isPausedForDamage = true
                setStatus("Sécurité (1min)")
                tpToSpawn()
                print("Dégâts reçus ! Téléportation au spawn et pause de 1 minute...")
                
                task.delay(60, function()
                    isPausedForDamage = false
                    print("Fin de la pause de sécurité. Reprise du farm.")
                end)
            end
            lastHealth = currentHealth
        end)
    end
end

if LocalPlayer.Character then listenToDamage(LocalPlayer.Character) end

local function getAxStats(axName, treClas)
    local module = AxeClasses:FindFirstChild("AxeClass_" .. axName)
    if not module then return {Damage = 2, SwingCooldown = 1} end
    local axStats = require(module).new()
    if axStats.SpecialTrees and axStats.SpecialTrees[treClas] then
        for k, v in next, axStats.SpecialTrees[treClas] do
            axStats[k] = v
        end
    end
    return axStats
end

local function getBestAxe(treClas)
    local axs = {}
    for _, tool in next, LocalPlayer.Backpack:GetChildren() do
        if tool:FindFirstChild("ToolName") and tool.Name ~= "BlueprintTool" then
            table.insert(axs, {ax = tool, stats = getAxStats(tool.ToolName.Value, treClas)})
        end
    end
    for _, tool in next, LocalPlayer.Character:GetChildren() do
        if tool:IsA("Tool") and tool:FindFirstChild("ToolName") and tool.Name ~= "BlueprintTool" then
            table.insert(axs, {ax = tool, stats = getAxStats(tool.ToolName.Value, treClas)})
        end
    end
    if #axs == 0 then return nil, nil end
    table.sort(axs, function(a, b) return a.stats.Damage > b.stats.Damage end)
    return axs[1].ax, axs[1].stats
end

local function findTreeByClass(clas)
    for _, region in next, Workspace:GetChildren() do
        if region.Name == "TreeRegion" then
            for _, tree in next, region:GetChildren() do
                if tree:IsA("Model") and tree:FindFirstChild("CutEvent") then
                    local jsonClas = tree:FindFirstChild("TreeClass")
                    local owner = tree:FindFirstChild("Owner")
                    if owner and jsonClas and owner.Value == nil and jsonClas.Value == clas then
                        if tree:FindFirstChild("WoodSection") then
                            return tree
                        end
                    end
                end
            end
        end
    end
    return nil
end

-- ==========================================
-- SÉQUENCE PRINCIPALE (COUPAGE ET VENTE)
-- ==========================================
local function processTree(tree)
    local treeClass = tree.TreeClass.Value
    local ax, axStats = getBestAxe(treeClass)
    
    if not ax then
        warn("Aucune hache trouvée ! Désactivation du script.")
        _G.FarmEnabled = false
        updateToggleButtonState()
        setStatus("Aucune hache !")
        tpToSpawn()
        return
    end

    setStatus("Coupure bois (" .. treeClass .. ")")
    LocalPlayer.Character.Humanoid:UnequipTools()
    task.wait()

    local follow = RunService.Heartbeat:Connect(function()
        if tree and tree:FindFirstChild("WoodSection") and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and not isPausedForDamage then
            LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(tree.WoodSection.Position + Vector3.new(0, 2, 0))
        end
    end)

    local newLog = nil
    local connection = Workspace.LogModels.ChildAdded:Connect(function(child)
        task.wait()
        if child:FindFirstChild("Owner") and child.Owner.Value == LocalPlayer then
            newLog = child
        end
    end)

    local cutProperties = {
        tool = ax,
        height = 0.3,
        faceVector = Vector3.new(1, 0, 0),
        sectionId = 1,
        hitPoints = axStats.Damage,
        cooldown = axStats.SwingCooldown,
        cuttingClass = "Axe"
    }

    repeat
        if not tree or not tree:FindFirstChild("CutEvent") then break end
        if isPausedForDamage or not _G.FarmEnabled then break end
        RemoteProxy:FireServer(tree.CutEvent, cutProperties)
        task.wait(axStats.SwingCooldown)
    until newLog ~= nil or not _G.FarmEnabled or isPausedForDamage

    if follow then follow:Disconnect() end
    if connection then connection:Disconnect() end

    if not _G.FarmEnabled or isPausedForDamage or not newLog then return end

    setStatus("Attente (5s)")
    task.wait(5)
    
    if isPausedForDamage or not _G.FarmEnabled then return end

    -- VENTE VIA KRONHUB
    setStatus("Vente automatique")
    loadstring(game:HttpGet('https://raw.githubusercontent.com/DevKron/Kron_Hub/refs/heads/main/lt2new'))("")
    
    local success, err = pcall(function()
        local mainHolder = CoreGui:WaitForChild("KronHub", 10):WaitForChild("Frame", 5):WaitForChild("MainHolder", 5)
        local woodTabs = mainHolder.BackGround.WoodTab:GetDescendants()
        
        for _, woodTab in ipairs(woodTabs) do
            if woodTab.Name ~= "ClickBt" then continue end
            local clickBt = woodTab
            
            if clickBt and clickBt:IsA("TextButton") then
                local text = tostring(clickBt.Text)

                if text == "Sell" then
                    firesignal(clickBt.MouseButton1Click)
                    print("Bouton Sell cliqué :", clickBt:GetFullName())
                    
                    local notify = CoreGui:WaitForChild("Notify", 120)
                    if not notify then return end
                    
                    local notifyFrame = notify:WaitForChild("Frame", 10)
                    if not notifyFrame then return end
                    
                    local sold = true
                    while sold and _G.FarmEnabled and not isPausedForDamage do
                        task.wait(1)
                        local notifs = notifyFrame:GetDescendants()
                        for _, notif in ipairs(notifs) do
                            if notif:IsA("TextLabel") or notif:IsA("TextButton") then
                                if notif.Text == "Done" then
                                    print("Vente terminée avec succès !")
                                    sold = false
                                    break
                                end
                            end
                        end
                    end
                    return 
                end
            end
        end
    end)

    if not success then
        warn("Erreur KronHub Sell:", err)
    end

    -- Incrémentation et mise à jour du compteur visuel
    totalTreesFarmedThisSession = totalTreesFarmedThisSession + 1
    updateTreeCounter()

    -- Vérification du Server Hop Quota
    if totalTreesFarmedThisSession >= _G.TreesBeforeHop then
        print("Quota total de session atteint (".._G.TreesBeforeHop.." arbres). Changement de serveur...")
        serverHop()
        return
    end

    task.wait(2)
end
