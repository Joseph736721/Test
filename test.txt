-- STOCKAGE ET FONCTIONS DE BASE
-- ==========================================
local treeregins = {}
local treclases = {}

for _, v in next, workspace:GetChildren() do
    if v.Name == "TreeRegion" then
        table.insert(treeregins, v)
    end
end

for _, region in next, treeregins do
    for _, tree in next, region:GetChildren() do
        if tree:FindFirstChild("TreeClass") and not table.find(treclases, tree.TreeClass.Value) then
            table.insert(treclases, tree.TreeClass.Value)
        end
    end
end

function getAxStats(axName, treClas)
    local module = axeclasses:FindFirstChild("AxeClass_" .. axName)
    if not module then return end

    local axStats = require(module).new()

    if axStats.SpecialTrees and axStats.SpecialTrees[treClas] then
        for k, v in next, axStats.SpecialTrees[treClas] do
            axStats[k] = v
        end
    end

    return axStats
end

function getModlMass(model)
    local total = 0
    local sections = 0

    for _, v in next, model:GetDescendants() do
        if v:IsA("BasePart") and v.Name ~= "Leaves" then
            if v.Name == "WoodSection" then
                sections += 1
            end
            total += v.Mass
        end
    end

    return total, sections
end

-- RECHERCHE LE MEILLEUR ARBRE SELON UNE CLASSE
function getTrreOfClas(clas)
    local options = {}

    for _, region in next, treeregins do
        for _, tree in next, region:GetChildren() do
            if tree:IsA("Model") and tree:FindFirstChild("CutEvent") then
                local jsonClas = tree:FindFirstChild("TreeClass")
                local owner = tree:FindFirstChild("Owner")

                if owner and jsonClas and owner.Value == nil and jsonClas.Value == clas then
                    local mass, secs = getModlMass(tree)
                    if secs > 1 then
                        table.insert(options, {tre = tree, mass = mass})
                    end
                end
            end
        end
    end

    table.sort(options, function(a, b)
        return a.mass > b.mass
    end)

    return options[1] and options[1].tre or false
end

-- TROUVE N'IMPORTE QUEL ARBRE DISPONIBLE ALÉATOIREMENT
function getRandomAvailableTree()
    local options = {}

    for _, region in next, treeregins do
        for _, tree in next, region:GetChildren() do
            if tree:IsA("Model") and tree:FindFirstChild("CutEvent") then
                local owner = tree:FindFirstChild("Owner")
                local jsonClas = tree:FindFirstChild("TreeClass")
                if owner and owner.Value == nil and jsonClas and tree:FindFirstChild("WoodSection") then
                    local _, secs = getModlMass(tree)
                    if secs > 1 then
                        table.insert(options, tree)
                    end
                end
            end
        end
    end

    if #options > 0 then
        return options[math.random(1, #options)]
    end
    return false
end

function choppy(tre, ax)
    task.wait()

    if not tre or not tre:FindFirstChild("WoodSection") then return nil end
    cooper.Character.HumanoidRootPart.CFrame = CFrame.new(tre.WoodSection.Position)

    task.wait(0.25)

    local axStats = getAxStats(ax.ToolName.Value, tre.TreeClass.Value)

    local propertys = {
        tool = ax,
        height = 0.3,
        faceVector = Vector3.new(1, 0, 0),
        sectionId = 1,
        hitPoints = axStats.Damage,
        cooldown = axStats.SwingCooldown,
        cuttingClass = "Axe"
    }

    local newtre = nil
    local conn

    conn = workspace.LogModels.ChildAdded:Connect(function(t)
        task.wait()
        if t:FindFirstChild("Owner") and t.Owner.Value == cooper then
            newtre = t
            conn:Disconnect()
        end
    end)

    repeat
        if not tre or not tre:FindFirstChild("CutEvent") then break end
        replicatedfarts.Interaction.RemoteProxy:FireServer(tre.CutEvent, propertys)
        task.wait(axStats.SwingCooldown)
    until newtre ~= nil or not _G.FarmEnabled

    if conn then conn:Disconnect() end
    return newtre
end

-- ==========================================
-- FONCTION PRINCIPALE (FARM SANS RETOUR)
-- ==========================================
function bringy(tre)
    if not tre or not tre:FindFirstChild("TreeClass") then return false, "Invalid tree" end
    local treclas = tre.TreeClass.Value

    cooper.Character.Humanoid:UnequipTools()
    task.wait()

    local axs = {}
    for _, tool in next, cooper.Backpack:GetChildren() do
        if tool:FindFirstChild("ToolName") and tool.Name ~= "BlueprintTool" then
            table.insert(axs, {
                ax = tool,
                stats = getAxStats(tool.ToolName.Value, treclas)
            })
        end
    end

    if #axs == 0 then
        return false, "No axe found"
    end

    table.sort(axs, function(a, b)
        return a.stats.Damage > b.stats.Damage
    end)

    local ax = axs[1].ax
    local follow

    if treclas ~= "LoneCave" then
        follow = RunService.Heartbeat:Connect(function()
            if tre and tre:FindFirstChild("WoodSection") then
                cooper.Character.HumanoidRootPart.CFrame = CFrame.new(tre.WoodSection.Position)
            end
        end)
    end

    tre = choppy(tre, ax)

    if follow then
        follow:Disconnect()
    end

    if not tre then 
        return false, "Chop interrupted or failed" 
    end

    task.wait(0.3)

    -- Coordonnées cibles pour chaque morceau de bois (Log)
    local targetLogPos = CFrame.new(315.2, 1, 100)

    local logsToTeleport = {}
    for _, child in next, tre:GetDescendants() do
        if child:IsA("BasePart") and child.Name == "WoodSection" then
            table.insert(logsToTeleport, child)
        end
    end

    -- Téléporte chaque bout de bois individuellement
    for _, log in next, logsToTeleport do
        task.spawn(function()
            for i = 1, 20 do
                replicatedfarts.Interaction.ClientIsDragging:FireServer(log)
                task.wait()
            end
        end)
        
        for i = 1, 15 do
            log.Velocity = Vector3.zero
            log.RotVelocity = Vector3.zero
            log.CFrame = targetLogPos
            task.wait()
        end
    end

    return true, "Done"
end

-- ==========================================
-- BOUCLE DE SÉLECTION INTELLIGENTE
-- ==========================================
task.spawn(function()
    while true do
        task.wait(0.5)
        
        if _G.FarmEnabled then
            local targetTree = nil
            
            -- 1. Priorité absolue : Volcano
            targetTree = getTrreOfClas("Volcano")
            
            -- 2. Deuxième priorité : Ice
            if not targetTree then
                targetTree = getTrreOfClas("Ice")
            end
            
            -- 3. Troisième priorité : N'importe quel arbre restant
            if not targetTree then
                targetTree = getRandomAvailableTree()
            end

            if targetTree then
                local success, msg = bringy(targetTree)
                if not success then
                    warn("[FARM FAIL]", msg)
                    task.wait(2)
                else
                    print("[FARM OK] Tree processed successfully")
                end
            else
                -- Aucun arbre du tout sur la map
                print("[FARM] Plus aucun arbre disponible sur la carte. Attente...")
                task.wait(3)
            end
        else
            task.wait(1)
        end
    end
end)
