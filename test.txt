-- PLAYER / SERVICES
local cooper = game.Players.LocalPlayer
local replicatedfarts = game.ReplicatedStorage
local axeclasses = replicatedfarts:WaitForChild("AxeClasses")

-- STORAGE
local treeregins = {}
local treclases = {}

-- FIND TREE REGIONS
for _, v in next, workspace:GetChildren() do
	if v.Name == "TreeRegion" then
		table.insert(treeregins, v)
	end
end

-- COLLECT TREE CLASSES
for _, region in next, treeregins do
	for _, tree in next, region:GetChildren() do
		if tree:FindFirstChild("TreeClass") and not table.find(treclases, tree.TreeClass.Value) then
			table.insert(treclases, tree.TreeClass.Value)
		end
	end
end

-- AXE STATS
function getAxStats(axName, treClas)
	local module = axeclasses:FindFirstChild("AxeClass_" .. axName)
	if not module then return end

	local axStats = require(module).new()

	if axStats.SpecialTrees and axStats.SpecialTrees[treClas] then
		for k, v in next, axStats.SpecialTrees[treClas] do
			axStats[k] = v
		end
	end

	return axStats
end

-- TREE MASS CHECK
function getModlMass(model)
	local total = 0
	local sections = 0

	for _, v in next, model:GetDescendants() do
		if v:IsA("BasePart") and v.Name ~= "Leaves" then
			if v.Name == "WoodSection" then
				sections += 1
			end
			total += v.Mass
		end
	end

	return total, sections
end

-- FIND BEST TREE
function getTrreOfClas(clas)
	local options = {}

	for _, region in next, treeregins do
		for _, tree in next, region:GetChildren() do
			if tree:IsA("Model") and tree:FindFirstChild("CutEvent") then
				local trreclas = tree:FindFirstChild("TreeClass")
				local owner = tree:FindFirstChild("Owner")

				if owner and trreclas and owner.Value == nil and trreclas.Value == clas then
					local mass, secs = getModlMass(tree)
					if secs > 1 then
						table.insert(options, {tre = tree, mass = mass})
					end
				end
			end
		end
	end

	table.sort(options, function(a, b)
		return a.mass > b.mass
	end)

	return options[1] and options[1].tre or false, "No tree found."
end

-- CHOP FUNCTION
function choppy(tre, ax)
	task.wait()

	cooper.Character.HumanoidRootPart.CFrame =
		CFrame.new(tre.WoodSection.Position + Vector3.new(5, 0, 0))

	task.wait(0.25)

	local axStats = getAxStats(ax.ToolName.Value, tre.TreeClass.Value)

	local propertys = {
		tool = ax,
		height = 0.3,
		faceVector = Vector3.new(1, 0, 0),
		sectionId = 1,
		hitPoints = axStats.Damage,
		cooldown = axStats.SwingCooldown,
		cuttingClass = "Axe"
	}

	local newtre = nil
	local conn

	conn = workspace.LogModels.ChildAdded:Connect(function(t)
		task.wait()
		if t:FindFirstChild("Owner") and t.Owner.Value == cooper then
			newtre = t
			conn:Disconnect()
		end
	end)

	repeat
		replicatedfarts.Interaction.RemoteProxy:FireServer(tre.CutEvent, propertys)
		task.wait(axStats.SwingCooldown)
	until newtre ~= nil

	return newtre
end

-- MAIN FARM FUNCTION
function bringy(treclas)
	local startpos = cooper.Character.HumanoidRootPart.CFrame
	cooper.Character.Humanoid:UnequipTools()
	task.wait()

	local axs = {}

	for _, tool in next, cooper.Backpack:GetChildren() do
		if tool:FindFirstChild("ToolName") and tool.Name ~= "BlueprintTool" then
			table.insert(axs, {
				ax = tool,
				stats = getAxStats(tool.ToolName.Value, treclas)
			})
		end
	end

	if #axs == 0 then
		return false, "No axe found"
	end

	table.sort(axs, function(a, b)
		return a.stats.Damage > b.stats.Damage
	end)

	local ax = axs[1].ax
	local tre = getTrreOfClas(treclas)

	if not tre then
		return false, "No tree found"
	end

	local follow

	if treclas ~= "LoneCave" then
		follow = game:GetService("RunService").Heartbeat:Connect(function()
			cooper.Character.HumanoidRootPart.CFrame =
				CFrame.new(tre.WoodSection.Position + Vector3.new(5, 3, 0))
		end)
	end

	tre = choppy(tre, ax)

	if follow then
		follow:Disconnect()
	end

	task.wait(0.2)

	task.spawn(function()
		for i = 1, 60 do
			replicatedfarts.Interaction.ClientIsDragging:FireServer(tre)
			task.wait()
		end
	end)

	tre.PrimaryPart = tre.WoodSection

	for i = 1, 60 do
		tre.PrimaryPart.Velocity = Vector3.zero
		tre:PivotTo(startpos)
		task.wait()
	end

	cooper.Character.HumanoidRootPart.CFrame = tre.WoodSection.CFrame

	return true, "Done"
end

-- AUTO FARM LOOP (VOLCANO)
local TARGET = "Volcano"

task.spawn(function()
	while task.wait(3) do
		local success, msg = bringy(TARGET)

		if not success then
			warn("[FARM FAIL]", msg)
			task.wait(5)
		else
			print("[FARM OK] Volcano chopped")
		end
	end
end)
